CREATE TABLE EMP(
 EMPNO NUMBER PRIMARY KEY,
 ENAME VARCHAR2(20),
 SAL NUMBER,
 HIREDATE DATE
);

SELECT SYSDATE AS "DATE"
FROM dual;


SELECT EMPNO,
       ENAME,
       SAL,
       ROUND(SAL * 1.15) AS "New Salary"
FROM EMP;


SELECT INITCAP(ENAME) AS "Name",
       LENGTH(ENAME) AS "Name Length"
FROM EMP;


SELECT ENAME,
       MONTHS_BETWEEN(SYSDATE, HIREDATE) AS "MONTHS_WORKED"
FROM EMP
ORDER BY MONTHS_WORKED;

create table inventory_master1
(
orderid number primary key,
custid number,
orderdate date,
amount number
);


create table inventory_trans1
(
orderid number references inventory_master1(orderid),
productid number,
productname varchar2(30),
quantity number,
unitprice number
);



create or replace trigger check1
before insert on inventory_master1
for each row
declare
begin

if(:new.amount < 0) then
raise_application_error(-20040,'Invalid Amount');
end if;

end;
/



insert into inventory_master1
values(1,101,'10-MAR-2026',5000);


create or replace trigger check2
before insert on inventory_trans1
for each row
declare
begin

if(:new.quantity < 0) then
raise_application_error(-20040,'Invalid Quantity');
end if;

end;
/




insert into inventory_trans1
values(1,201,'Laptop',2,50000);


select * from inventory_master1;

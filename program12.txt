SET SERVEROUTPUT ON;

CREATE TABLE bank_acc(
accno VARCHAR2(10),
name VARCHAR2(20),
deposit NUMBER
);

DECLARE
vaccno VARCHAR2(10):='&vaccno';
vname VARCHAR2(20):='&vname';
vdeposit NUMBER:=&vdeposit;
zerodeposit EXCEPTION;

BEGIN

IF vdeposit<=0 THEN
RAISE zerodeposit;

ELSE
INSERT INTO bank_acc(accno,name,deposit)
VALUES(vaccno,vname,vdeposit);

DBMS_OUTPUT.PUT_LINE('Record is successfully inserted');

END IF;

EXCEPTION
WHEN zerodeposit THEN
DBMS_OUTPUT.PUT_LINE('Invalid deposit amount');

END;
